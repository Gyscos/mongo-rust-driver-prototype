macro_rules! separated_int {
    ($string:expr) => {{
        let mut output = String::new();
        let magnitude = if $string.starts_with('-') {
            output.push('-');
            (&$string)[1..].to_owned()
        } else {
            $string
        };

        separated_uint_with_output!(magnitude, output)
    }};
}
